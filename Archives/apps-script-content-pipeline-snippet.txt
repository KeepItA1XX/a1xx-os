// Add this function near getNotionContent(e):
function getContentPipeline(e) {
  var secret = PropertiesService.getScriptProperties().getProperty('NOTION_SECRET');
  var databaseId = e.parameter.databaseId || NOTION_CONTENT_DB;

  if (!secret) return error('NOTION_SECRET not set in Script Properties.');

  var res = UrlFetchApp.fetch(
    'https://api.notion.com/v1/databases/' + databaseId + '/query',
    {
      method: 'post',
      headers: {
        Authorization: 'Bearer ' + secret,
        'Notion-Version': '2022-06-28',
        'Content-Type': 'application/json'
      },
      payload: JSON.stringify({
        filter: {
          property: 'Status',
          status: {
            does_not_equal: 'Posted'
          }
        },
        sorts: [
          {
            property: 'Priority',
            direction: 'ascending'
          },
          {
            timestamp: 'last_edited_time',
            direction: 'descending'
          }
        ],
        page_size: 50
      }),
      muteHttpExceptions: true
    }
  );

  var text = res.getContentText();
  var code = res.getResponseCode();

  logActivity('Content pipeline request — code: ' + code);

  if (code >= 400) {
    return error('Notion content pipeline ' + code + ': ' + text);
  }

  return ContentService
    .createTextOutput(text)
    .setMimeType(ContentService.MimeType.JSON);
}
